pragma solidity ^0.8.0;

contract LandRegistration {

    struct Land {
        string state;
        string district;
        string village;
        uint surveyNumber;
        address owner;
    }

    mapping(uint => Land) public lands;

    event LandRegistered(uint id, address owner);

    // Register land
    function registerLand(
        uint id,
        string memory _state,
        string memory _district,
        string memory _village,
        uint _surveyNumber
    ) public {
        lands[id] = Land(_state, _district, _village, _surveyNumber, msg.sender);
        emit LandRegistered(id, msg.sender);
    }

    // Get land details
    function getLand(uint id) public view returns (
        string memory, string memory, string memory, uint, address
    ) {
        Land memory l = lands[id];
        return (l.state, l.district, l.village, l.surveyNumber, l.owner);
    }
}

async function connectWallet() {
    if (window.ethereum) {
        try {
            provider = new ethers.ethers.providers.Web3Provider(window.ethereum);


            await provider.send("eth_requestAccounts", []);


            signer = provider.getSigner();


            let address = await signer.getAddress();
            document.getElementById("account").innerText = "Connected: " + address;


            contract = new ethers.ethers.Contract(contractAddress, abi, signer);


        } catch (err) {
            console.error(err);
            alert("Connection failed");
        }
    } else {
        alert("Install MetaMask!");
    }
}
